    public void fibonacci( int n) {

        int sum = 0;

        int sumfin = 0;

        for(int i = 1; i <= n; i++) {

            for(int j = 2; j > i ; j++) {

                if(sumfin > sum) {

                    System.out.print(sum + sumfin);

                } else {

                    break;

                }

                sum = j + i;

                sumfin = sum + i;

 

            }

        }

    }
